\begin{block}{Coordinate transformation}
system with basis
\begin{equation*}
B_{BAD} = \left(
\begin{array}{ccc}
\vec{n} & \vec{AB} & \vec{AD}
\end{array}
\right)
\end{equation*}
yields
\begin{equation*}
\left(B_{BAD}\right)^{-1} P_1
=
\left(
\begin{array}{ccc}
d&u&v
\end{array}
\right)^T
\end{equation*}
\end{block}
}
\only<3>{
\begin{block}{Problem:}
\begin{itemize}
\item[\textcolor{green}{\Checkmark}] for $P_1$: $\left(u,v\right) = \left(0.5,0.4\right)$
\item[\textcolor{red}{\XSolidBrush}] for $P_2$: $\left(u,v\right) = \left(1,1\right)$
\end{itemize}
\end{block}

\begin{block}{Solution:}
\begin{enumerate}
\item if we get $u+v > 1$
\item use $B_{BCD}$ instead of $B_{BAD}$
\item set $u=1-u$, $v=1-v$
\end{enumerate}
\end{block}
